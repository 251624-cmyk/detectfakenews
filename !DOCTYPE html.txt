let index=0;
let score=0;

const card=document.getElementById("card");
const headline=document.getElementById("headline");
const imageArea=document.getElementById("imageArea");
const reason=document.getElementById("reason");
const nextBtn=document.getElementById("nextBtn");

const realBtn=document.getElementById("realBtn");
const fakeBtn=document.getElementById("fakeBtn");

function loadCard(){

    card.className="card";

    reason.style.display="none";
    nextBtn.style.display="none";

    realBtn.disabled=false;
    fakeBtn.disabled=false;

    const item=newsData[index];

    headline.innerText=item.headline;

    if(item.image){
        imageArea.innerHTML=`<img src="${item.image}" alt="">`;
    }else{
        imageArea.innerHTML="";
    }

}

function answer(userFake){

    realBtn.disabled=true;
    fakeBtn.disabled=true;

    const item=newsData[index];

    const correct=(userFake===item.isFake);

    if(correct){
        score++;
        card.classList.add("correct");
    }else{
        card.classList.add("wrong");
    }

    reason.style.display="block";
    reason.innerHTML=
        "<b>"+(correct?"정답!":"오답!")+"</b><br><br>"+item.reason;

    nextBtn.style.display="block";

}

realBtn.onclick=()=>answer(false);
fakeBtn.onclick=()=>answer(true);

nextBtn.onclick=()=>{

    index++;

    if(index>=newsData.length){

        card.innerHTML=`
        <div class="score">
            <h1>🎉 게임 종료!</h1>
            <p><b>${newsData.length}개 중 ${score}개</b>를 맞혔습니다.</p>
            <br>
            <button onclick="location.reload()">다시 하기</button>
        </div>
        `;

        return;
    }

    loadCard();

};

loadCard();
